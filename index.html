        // Обновление статистики

        function updateStats() {

            // Обновляем онлайн

            const onlineCount = Math.floor(Math.random() * 5000) + 1000;

            document.getElementById('online-count').textContent = onlineCount;

            document.getElementById('current-online').textContent = onlineCount;

            

            // Рассчитываем нагрузку на сервер (0-100%)

            const serverLoad = Math.min(100, Math.floor(onlineCount / 60));

            document.getElementById('server-load').textContent = serverLoad + '%';

            document.getElementById('current-load').textContent = serverLoad + '%';

            

            // Рассчитываем количество отправленных сообщений

            const messagesSent = Math.floor(onlineCount * 0.05 + Math.random() * 20);

            document.getElementById('messages-sent').textContent = messagesSent;

            document.getElementById('today-messages').textContent = Math.floor(messagesSent * 0.8 + Math.random() * 10);

            

            // Обновляем графики

            appData.onlineHistory.push(onlineCount);

            appData.onlineHistory.shift();

            

            appData.loadHistory.push(serverLoad);

            appData.loadHistory.shift();

            

            appData.messagesHistory.push(messagesSent);

            appData.messagesHistory.shift();

            

            onlineChart.update();

            loadChart.update();

            messagesChart.update();

        }

        function enableMobilePreview(width) {

            const container = document.getElementById('device-size-selector');

            const previewBtn = document.getElementById('mobile-preview-btn');

            

            if (width === '100%') {

                appData.mobileView = false;

            } else {

                document.body.style.width = width + 'px';

                document.body.style.maxWidth = width + 'px';

                document.body.style.margin = '0 auto';

                appData.mobileView = true;

            }

        // Создаем различные фигуры

        function createShapes() {

            sceneContainer.innerHTML = '';

            shapes.length = 0;

            

            if (!appData.shapesEnabled) return;

            

            const shapesData = [

                { type: 'cube', class: 'cube' },

                { type: 'sphere', class: 'sphere' },

                { type: 'pyramid', class: 'pyramid' },

                { type: 'torus', class: 'torus' }

            ];

            

            // Создаем 10 фигур случайных типов

            for (let i = 0; i < 10; i++) {

                const randomShape = shapesData[Math.floor(Math.random() * shapesData.length)];

                const shape = document.createElement('div');

                shape.className = `shape ${randomShape.class}`;

                

                // Случайное позиционирование

                const x = Math.random() * 100;

                const y = Math.random() * 100;

                

                shape.style.left = `${x}vw`;

                shape.style.top = `${y}vh`;

                

                // Случайная анимация плавания

                shape.style.animation = `float ${6 + Math.random() * 10}s ease-in-out infinite`;

                shape.style.animationDelay = `${Math.random() * 5}s`;

                

                sceneContainer.appendChild(shape);

                shapes.push({

                    element: shape,

                    x: x,

                    y: y,

                    rotation: 0

                });

            }

        }

        

        // Обработка движения мыши для параллакс-эффекта

        function handleMouseMove(e) {

            const mouseX = e.clientX / window.innerWidth - 0.5;

            const mouseY = e.clientY / window.innerHeight - 0.5;

            

            shapes.forEach(shape => {

                const moveX = mouseX * 20;

                const moveY = mouseY * 20;

                

                shape.element.style.transform = `translate(${moveX}px, ${moveY}px) rotate(${shape.rotation}deg)`;

            });

        }

        

        // Обработка скролла для вращения фигур

        function handleScroll() {

            const scrollY = window.scrollY;

            

            shapes.forEach(shape => {

                shape.rotation = scrollY / 20;

                shape.element.style.transform = `rotate(${shape.rotation}deg)`;

            });

        }

        function createNewChat(name) {

        function loadChat(chatId) {

            appData.currentChat = chatId;

            messagesContainer.innerHTML = '';

        function sendMessage() {

            const input = document.getElementById('message-input');

            const message = input.value.trim();

            

            if (message && appData.currentChat) {

                // Получаем текущую нагрузку на сервер

                const serverLoad = parseInt(document.getElementById('server-load').textContent);

                

                // Если нагрузка высокая, с шансом 30% показываем сообщение об ошибке

                if (serverLoad > 70 && Math.random() < 0.3) {

                    const notification = document.getElementById('high-load-notification');

                    notification.style.display = 'block';

                    

                    setTimeout(() => {

                        notification.style.display = 'none';

                    }, 3000);

                    

                    return;

                }

                

                // Добавляем исходящее сообщение

                addMessageToChat(message, 'outgoing');

                appData.messages[appData.currentChat].push({

                    text: message,

                    type: 'outgoing'

                });

                

                // Очищаем поле ввода

                input.value = '';

                

                // Проверяем шаблоны ИИ для ответа

                setTimeout(() => {

                    let response = "Извините, я не понимаю. Добавьте шаблоны в настройках обучения ИИ.";

                    

                    for (const pattern of appData.aiPatterns) {

                        if (pattern.question.toLowerCase() === message.toLowerCase()) {

                            response = pattern.answer;

                            break;

                        }

                    }

                    

                    // Добавляем входящее сообщение (ответ ИИ)

                    addMessageToChat(response, 'incoming');

                    appData.messages[appData.currentChat].push({

                        text: response,

                        type: 'incoming'

                    });

                    

                    // Сохраняем данные

                    saveData();

                }, 1000);

            }

        }

        function addPattern() {

            const input = document.getElementById('pattern-input');

            const patternText = input.value.trim();

            

            if (patternText.includes('>')) {

                const parts = patternText.split('>');

                if (parts.length >= 2) {

                    const question = parts[0].trim();

                    const answer = parts[1].trim();

                    

                    if (question && answer) {

                        // Добавляем шаблон в массив

                        appData.aiPatterns.push({ question, answer });

                        

                        // Добавляем шаблон в список

                        addPatternToUI(question, answer);

                        

                        // Очищаем поле ввода

                        input.value = '';

                        

                        // Сохраняем данные

                        saveData();

                    }

                }

            }

        }

        

        function addPatternToUI(question, answer) {

            const patternItem = document.createElement('div');

            patternItem.className = 'pattern-item';

            

            const patternText = document.createElement('span');

            patternText.textContent = `${question} → ${answer}`;

            

            const deleteBtn = document.createElement('button');

            deleteBtn.className = 'delete-pattern';

            deleteBtn.textContent = 'Удалить';

            deleteBtn.addEventListener('click', function() {

                const index = appData.aiPatterns.findIndex(p => p.question === question && p.answer === answer);

                if (index !== -1) {

                    appData.aiPatterns.splice(index, 1);

                }

                patternItem.remove();

                saveData();

            });

            

            patternItem.appendChild(patternText);

            patternItem.appendChild(deleteBtn);

            document.getElementById('patterns-list').appendChild(patternItem);

        }
